function binarySearchDictionary(wordList, word) {
  var start = 0;
  var end = wordList.length - 1;

  while (start <= end) {
    var middle = Math.round((start + end) / 2);
    if (wordList[middle] > word) {
      var end = middle - 1;
    } else if (wordList[middle] < word) {
      var start = middle + 1;
    } else {
      return middle;
    }
  }
  return "Word not found";
}

function findGoldfish(fishInTank, whiteGoldfish) {
    var winningGoldfish = null;
    var chosenGoldfish;

    for(i = 0; i < fishInTank.length; i ++) {
        chosenGoldfish = fishInTank[i];
        if (chosenGoldfish === whiteGoldfish ) {
            winningGoldfish = i;
            break;
        }
    }
    return winningGoldfish;
}

findGoldfish(['clown fish', 'molly', 'tetra', 'white goldfish', 'guppy', 'fighting fish'], 'white goldfish');
